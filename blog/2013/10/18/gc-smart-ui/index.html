行为树实现

Task = {
    update = function(args) end,
    on_enter = function(args) end,
    on_exit = function(args) end,
}
BehaviorNode = {
    nodes = {},
    status = 0,
    task = nil,
    tick = function(args)
        return task:update(args)
    end,
}
BehaviorTree = {
    object = nil,
    root = nil,
    tick = function(args)
        if root then
            root:tick(args)
        end
    end,
}
